Debian GNU/Linux is a particular distribution of the Linux operating system, and numerous packages that run on it.

# Settings
```bash
sudo apt update
sudo apt purge vim-tiny
sudo apt install -y \
  bash-completion \
  curl \
  g++ \
  gcc \
  git \
  libbz2-dev \
  libffi-dev \
  libfreetype6-dev \
  liblzma-dev \
  libpq-dev \
  libreadline-dev \
  libsqlite3-dev \
  libssl-dev \
  llvm-dev \
  make \
  pkg-config \
  vim \
  zlib1g-dev \

```
